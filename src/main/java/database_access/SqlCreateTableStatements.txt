drop table if exists users;
drop table if exists persons;
drop table if exists events;
drop table if exists authTokens;

create table users
(
    username varchar(255) not null primary key,
    password varchar(255) not null,
    email varchar(255) not null,
    firstName varchar(255) not null,
    lastName varchar(255) not null,
    gender char(1) in ("f","m") not null,
    foreign key(personID) references persons(personID) not null
);

create table persons
(
    personID varchar(255) not null primary key,
    foreign key(descendant) references users(username) not null,
    firstName varchar(255) not null,
    lastName varchar(255) not null,
    gender char(1) in ("f","m") not null,
    foreign key(father) references persons(personID),
    foreign key(mother) references persons(personID),
    foreign key(spouse) references persons(personID),
);

create table events
(
    eventID varchar(255) not null primary key,
    foreign key(descendant) references users(username) not null,
    foreign key(person) references persons(personID) not null,
    latitude real not null,
    longitude real not null,
    country varchar(255) not null,
    city varchar(255) not null,
    eventType varchar(255) not null,
    year integer not null
);

create table authTokens
(
    authToken varchar(255) not null primary key,
    foreign key(username) references users(username)
);
